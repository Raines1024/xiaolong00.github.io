Collections.sort(list, new Comparator<OmsCouponSend>() {
    @Override
    public int compare(OmsCouponSend o1, OmsCouponSend o2) {
        if (o1.getCoupon() > o2.getCoupon()) {
            return -1;
        }
        if (o1.getCoupon().equals(o2.getCoupon())) {
            return 0;
        }
        return 1;
    }
});
